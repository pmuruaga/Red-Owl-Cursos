CREATE PROCEDURE usp_Instructor_Editar(
	@InstructorId uniqueidentifier,
	@Nombre varchar(500),
	@Apellidos varchar(500),
	@Grado varchar(100)
)
AS
	BEGIN
		UPDATE Instructor
		SET
			Nombre = @Nombre,
			Apellido = @Apellidos,
			Grado = @Grado,
			FechaCreacion = GETUTCDATE()
		WHERE
			InstructorId = @InstructorId
	END
